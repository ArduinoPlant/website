#define LED_OUTPUT 13

                #define ON 1

                #define OFF 0

 

                int sensorValue;

                int pump;

 

                void setup() {

                        Serial.begin(9600);

                        // initialize digital pin LED_BUILTIN as an output.

                        pinMode(LED_OUTPUT, OUTPUT);

                        pump = OFF;

                }

 

                void loop() {

                 

                  delay(1000);

                  sensorValue = analogRead(A0);

                  Serial.println(sensorValue);

                  if (pump == ON) {

                        if (sensorValue < 700) {

                          pump = OFF;

                          digitalWrite(LED_OUTPUT, LOW);

                        }

                  }

                  else

                  {

                        if (sensorValue > 800) {

                          pump = ON;

                          digitalWrite(LED_OUTPUT, HIGH);

                        }

                  }

                }
